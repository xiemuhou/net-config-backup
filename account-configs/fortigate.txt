# FortiGate 只读备份账号示例
# 请把 <STRONG_PASSWORD> 替换为强密码。

config system accprofile
    edit "backup-readonly"
        set secfabgrp read
        set ftviewgrp read
        set authgrp read
        set sysgrp read
        set netgrp read
        set loggrp read
        set fwgrp read
        set vpngrp read
        set utmgrp read
        set wanoptgrp read
        set wifi read
    next
end

config system admin
    edit "backup"
        set password <STRONG_PASSWORD>
        set accprofile "backup-readonly"
        set vdom "root"
        set ssh-public-key1 ''
    next
end
